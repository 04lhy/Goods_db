## 二：支持六大基本语句：

*   `show tables;` //列出数据库中所有的表名
*   `select ...... from 表名 （ where 属性= 值）` // 仅支持 到基本语句（后面+and 不支持）
*   `delete from 表名 where 列=值`
*   `update 表名 set 列=值 where 列=值`
*   `insert的标准SQL语句：insert into 表名 values(...)`
*   `create的标准SQL语句：create table 表名(属性名 数据类型 {约束条件} {, ... })`
    *   示例：`create table students(snum char(10) primary key, sname char(30) not null, sage int, sgpa double)`

**仅支持char 和 int ，double三种数据类型**

### 1. create table
#### a. 函数流程
![create table流程图]
*流程图内容：语法分析 -> 通过/不通过报错 -> 记录表信息（表名、属性名、属性类型、约束种类） -> 创建三个文件：.dat(用于保存insert)、.frm(保存表头信息)、.idx(保存索引信息)。其中索引使用map实现。*

#### b. 文件格式解析
**.frm文件：**
![frm文件截图]
该文件的每一行保存了表的一个属性，如第一行：`name 2 20 1`  
表示：
*   属性名：name
*   属性类型为 char （0表示为int，1表示为double，2表示为char）
*   长度为20字节
*   1表示约束类型主键（0表示无约束，1表示主键，2表示非空，4表示unique）

**.dat文件：**
对应.frm文件中的字段顺序，保存了每条数据对应的值。

**.idx文件：**
根据.frm文件中的有约束的字段建立索引。无任何约束则为空。

```cpp
string insert_handler(string sql)
{
    string token, table_name, values = "";
    stringstream sstrm(sql);
    sstrm >> token;
    if (token != "insert")
        return(SYNTAX_ERROR);
    sstrm >> token;
    if (token != "into")
        return(SYNTAX_ERROR);
    sstrm >> table_name;
    if (tables_map.find(table_name) == tables_map.end())
        return("table " + table_name + " is not exist!\n");
    while (sstrm >> token)/* 获取查找关键字 */
    {
        values += token;
        values += " ";
    }
    Table* tb = tables_map[table_name]; /* 将指针指向对应的表格的管理结构table */
    string ret = tb->insert(values);
    return(ret);
}
```

## 四：持久化实现
持久化主要依靠文件实现，当需要对一个表做基本语句时，可直接使用该类中的API。
具体实现，参见代码文件。

```cpp
#ifndef TABLE_H
#define TABLE_H

#include <iostream>
#include <string>
#include <vector>
#include <map>
#include <fstream>
#include "index.h"
//#include "util.h"
using namespace std;

/* 数据类型 */
enum DataType
{
    Int, Double, String
};

/* 列（属性） */
struct Column {
    string column_name; /* 属性名 */
    DataType data_type; /* 数据类型 */
    int offset; /*在元组中的偏移量*/
    int data_size; /* 数据长度（字节） */
    int constraints; /* 约束条件 0x1 主键 0x2 not null 0x4 unique 可组合 */
    
    Column(string name, int type, int offst, int size, int cons)
    {
        column_name = name;
        data_type = (DataType)type;
        offset = offst;
        data_size = size;
        constraints = cons;
    }
};

/* 对表的操作类 */
class Table {
public:
    Table(string table_name); /*初始化*/
    ~Table();
    /* 增删查改 */
    string insert(string values);
    string erase(string key_value);
    string select(string key_value);
    string update(string new_key_value, string old_key_value);
    string getTableName();
    void show_table_info();

private:
    fstream file_frm, file_dat; /* fstream fout_frm_fout_dat; */
    string table_name; /* 表名 */
    string primary_attribute;/*主属性*/
    vector<Column> column; /* 属性列信息 */
    map<string, int> column_index; /*属性名映射到第n列*/
    int total_size; /*元组总长度*/
    int insert_pos; /*插入位置（数据文件末尾）*/
    Index* index; /*索引接口*/

    string check_insert_valid(vector<string>& values);//检查插入完整性
    void erase_help(int n);/*删除第n个元组*/
    bool update_help(string& tuple, string& key, string& value);//将传入的元组字符串的key列替换成value值
    vector<int>& search(string& key, string& val); //通用搜索接口
    vector<int>& select_list_traverse(string& key, string& val);/*根据key列找值为val的元组*/
    string& operator[](int n); /* 返回表中第n个元组 */
    string value(int n, int col); /* 返回元组n中的第col属性列的值 */
};
#endif
```

## 五：索引实现：
该类专门用于管理索引，需要建立或操作索引时，可直接调用其API。
具体实现，参见代码文件。

```cpp
#pragma once
#ifndef INDEX_H
#define INDEX_H
#include <iostream>
#include <string>
#include <vector>
#include <map>
#include <fstream>
using namespace std;

/* 索引管理结构*/
class Index {
public:
    Index(string table_name); /*初始化*/
    ~Index();
    /* 索引结构的增删查改 */
    bool insert(string key, int idx); /* 主键映射到下标，返回插入结果 */
    bool erase(string key); /*删除key */
    bool erase(int idx); /*删除value=idx的映射*/
    bool find(string key); /* 查询键值是否已经存在 */
    int find_idx(string key); /*查询主键对应的元组下标*/
    vector<int>& all_idx(); /*返回所有下标*/
private:
    fstream file_idx; /* 初始时加载索引文件的指针 */
    string path_idx; /* 索引文件路径 */
    map<string, int> mp; /*根据主键映射到所在的位置（主存储结构）*/
};
#endif
```
